class PoliticasAprox(Politica):
    '''
    Descripción:
        * Clase padre de todas las políticas de aproximación
    Atributos:
        * Hereda todo de la clase padre 'Política'
        * betas = lista con los betas a actualizar

    Métodos:
        * epsilon_greedy_policy_VFA
        * SGD
        * obtener_features
        * mejor_accion: sobreesccribe la mejor accion en las polticias tabulares
        * probar politica: sobreescribe probar politica de las tabulares
        * simular: paso

    '''
    def __init__(self, racetrack, episodes, gamma, epsilon, learning_rate, betas = [0,0,0]):
        super().__init__(racetrack, episodes,gamma, epsilon)
        self.betas = betas
        self.learning_rate = learning_rate

    def epsilon_greedy(self,state):
        '''
        Descripción:
            * Método que implementa epsilon greedy policy utilizando VF
        Parámetros:
            * state: estado actual en el que se está
        Return:
            * mejor_accion, features de ese estado-accion
        '''
        if np.random.random() < self.epsilon:
            action = random.choice(self.racetrack.actions)
            features = self.obtener_features(state, action)
            return action, features
        else:
            acciones = self.racetrack.actions #lista de tuplas
            mejor_Q = -np.inf
            mejor_accion = None #
            features = None
            for action in acciones: 
                x = self.obtener_features(state, action) # Lista de features
                Q = np.dot(x, self.betas) # Predicción
                if Q > mejor_Q:
                    mejor_Q = Q
                    mejor_accion = action
                    features = x
            return mejor_accion, features

    def SGD(self,x,y):
            '''
            Descripción:
            *   Método que aplica SGD, recibe un array x y actualiza los pesos originados self.betas
            Parámetros:
            *   x: Array con los features de ese estado acción
            *   y: Recompensa obtenida en ese estado acción
            '''
            # Compute the gradients
            prediction = np.dot(x, self.betas)
            error = prediction - y
            gradients = 2 * np.dot(x.T, error)
            # Update the parameters
            self.betas -= self.learning_rate * gradients

    def obtener_features(self, state, action):
            '''
                Descripción:
                *   Método que obtiene los features para un estado acción con tal de actualizar los valores Beta

                Parámetros:
                *   state: tupla con la información asociada al estado Ej: ((10,5),(1,-1))
                *   action: tupla con la Acción realizada Ej: (1,0)

                Retorna:
                *   tupla con los features que se buscan para ese estado acción
                *   
            '''

            y_post,x_post, vy_post,vx_post, estado_post = self.simular_paso(state,action)
            final = np.array([9,44])  # coordenadas del centro de la meta
            # Calcular la distancia entre la posición post decisión y el centro de la meta
            distancia = np.hypot( y_post -final[0] , x_post - final[1]) # feature 1

            # Determinamos la segunda feature, si sigue o no o llega
            estado = None
            if estado_post == True:
                estado = 1 # Sigue en la pista
                # Vemos si en el futuro se caerá o no
                casos = []
                for i in self.racetrack.actions:
                    _,_,_,_,secae = self.simular_paso(((y_post,x_post),(vy_post,vx_post)), i )
                    casos.append(secae)
                    if True not in casos:
                        estado = 0 # se caerá en el futuro

            elif estado_post == False:
                estado = 0 # Se fue

            # Ahora determinamos el producto punto entre un dos vectores, el de la dirección hacia la meta y la dirección de la velocidad
            posicion = np.array([y_post,x_post])
            vector_direccion_unitario = (final - posicion) / np.linalg.norm((final - posicion))
            vector_velocidad_unitario = np.array([vy_post, vx_post]) / np.linalg.norm(np.array([vy_post,vx_post])) 
            producto_punto = np.dot(vector_direccion_unitario,vector_velocidad_unitario) #feature 3

            return np.array([distancia, estado, producto_punto])

    def politica_optima(self,state):
            '''
            Descripción:
            *   Método que devuelve la mejor accion para un estado utilizando la función de valor
            Parámetros:
            *   state: Estado
            Return:
            *   action: Mejor acción
            '''
            
            acciones = self.racetrack.actions #lista de tuplas
            mejor_Q = -np.inf
            mejor_accion = None #random.choice(self.racetrack.actions)
            for action in acciones: 
                x = self.obtener_features(state, action) # Lista de features
                Q = np.dot(x, self.betas) # Predicción
                if Q > mejor_Q:
                    mejor_Q = Q
                    mejor_accion = action
            return mejor_accion

    def probar_politica(self):
        '''
        Descripción:
            * Método que prueba la política aprendida ejecutando un episodio.

        Parámetros:
            * None

        Return:
            * Trajectory: lista que contiene la trayectoria [estado, acción, recompensa].
        '''
        self.racetrack.reset()
        done = False
        trajectory = []
        state = self.racetrack.get_state()
        recomensafinal=0
        while not done:
            action = self.politica_optima(state)
            reward = self.step(action)
            recomensafinal+=reward
            trajectory.append([state, action, reward])
            state = self.racetrack.get_state()
            done = self.finished(state)
        print(recomensafinal)
            
        return trajectory

    def simular_paso(self,state,action):
        '''
        Descripción:
        *   Método que devuelve el estado en el que se encontrará la simulación al recibir un estado y acción
        
        Parámetros:
        *   state: Un estado ((y,x), (vy,vx))
        *   action: Acción tomada Ej: (1,1)
        '''
        x = state[0][1]
        y = state[0][0]
        vx = state[1][1]
        vy = state[1][0]
        accion_x = action[1]
        accion_y = action[0]

        v_y_temp = vy + accion_y
        v_x_temp = vx + accion_x

        estado = None # Representa si se salió o no de la pista

        if self.racetrack.v_min <= v_x_temp <= self.racetrack.v_max:
            vx = v_x_temp
        if self.racetrack.v_min <= v_y_temp <= self.racetrack.v_max:
            vy = v_y_temp

        # Asegurarse de que al menos una componente de la velocidad sea no nula
        if vx == 0 and vy == 0:
            vx = 1
        # Actualizamos las posiciones
        x += vx
        y += vy

        if self.racetrack.track[y, x] != '#' :
            estado = True # Verifica que esté en la pista
        else:
            estado = False
        return y,x,vy,vx,estado
